#inlude <stdio.h>
#include <conio.h>

long int fact(int n);
int main() {
	int n;
	printf("Enter a positive integer :");
	scanf("%n",&n);
	printf("Factorial of %d = %d",n,fact(n));
	return 0;
}
long int fact (int n) {
	if(n>=1)
		return n*fact(n-1);
	else
		return 1;
}
